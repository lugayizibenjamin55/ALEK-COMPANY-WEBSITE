            const userPrompt = projectIdeaInput.value.trim();
            if (!userPrompt) {
                showMessageBox("Please enter a project type to generate an idea.");
                return;
            }

            // Show loading spinner
            generateIdeaSpinner.classList.remove('hidden');
            generateIdeaButton.disabled = true;
            generatedIdeaOutput.classList.add('hidden'); // Hide previous output

            try {
                let chatHistory = [];
                const prompt = `Generate a unique and innovative construction project idea based on the following type: "${userPrompt}". Provide a project name and a brief, compelling description (around 3-5 sentences) of its key features and benefits, suitable for a company like Alek Techbuild. Format the response as a JSON object with "projectName" and "projectDescription" keys.`;
                chatHistory.push({ role: "user", parts: [{ text: prompt }] });

                const payload = {
                    contents: chatHistory,
                    generationConfig: {
                        responseMimeType: "application/json",
                        responseSchema: {
                            type: "OBJECT",
                            properties: {
                                "projectName": { "type": "STRING" },
                                "projectDescription": { "type": "STRING" }
                            },
                            "propertyOrdering": ["projectName", "projectDescription"]
                        }
                    }
                };

                const apiKey = ""; // Canvas will automatically provide the API key
                const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=${apiKey}`;

                const response = await fetch(apiUrl, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify(payload)
                });

                const result = await response.json();

                if (result.candidates && result.candidates.length > 0 &&
                    result.candidates[0].content && result.candidates[0].content.parts &&
                    result.candidates[0].content.parts.length > 0) {
                    const jsonString = result.candidates[0].content.parts[0].text;
                    const parsedJson = JSON.parse(jsonString);

                    generatedIdeaTitle.textContent = parsedJson.projectName || "Generated Project Idea";
                    generatedIdeaDescription.textContent = parsedJson.projectDescription || "No description available.";
                    generatedIdeaOutput.classList.remove('hidden');
                } else {
                    showMessageBox("Failed to generate project idea. Please try again.");
                }
            } catch (error) {
                console.error("Error generating project idea:", error);
                showMessageBox("An error occurred while generating the project idea. Please try again later.");
            } finally {
                // Hide loading spinner
                generateIdeaSpinner.classList.add('hidden');
                generateIdeaButton.disabled = false;
            }
